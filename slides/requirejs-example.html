  require(["app/data", "helper/logger"], function (data, logger) {
    logger('data[0] : ' + data[0]);
    logger('data[1] : ' + data[1]);
  });
This function is called when after the others are loaded.
If logger.js calls define(), then this function is not fired until loggers's dependencies have loaded, and the `logger` argument will hold the module value for "helper/logger".
